**递归**

```js
var inorderTraversal = function(root) {
  let array=[];

  function help(root){
    if(!root){
      return;                  
    }
    help(root.left);
    array.push(root.val);
    help(root.right);                  
  }
  
  help(root);
  return array;
};
```

**迭代**

```js
var inorderTraversal = function(root) {
    const res = [];
    const stk = [];
    while (root || stk.length) {
        while (root) {
            stk.push(root);
            root = root.left;
        }
        root = stk.pop();
        res.push(root.val);
        root = root.right;
    }
    return res;
};
```
