

1. **顺序合并**

```js
var mergeKLists = function(lists) {
    let  result=null;
    for(let i=0;i<lists.length;i++){
        result=mergeTwoLists(lists[i],result);
    }
    return result;
};

let mergeTwoLists = function(a,b){
    if(a===null) return b;
    if(b===null) return a;
    let result = new ListNode(0);
    let temp=result;
    while(a!==null && b!==null){
        if(a.val<b.val){
            temp.next=a;
            a=a.next;
        }else{
            temp.next=b;
            b=b.next;
        }
        temp=temp.next;
    }
    if(a===null){
        temp.next=b;
    }else{
        temp.next=a;
    }
    return result.next;
}
```

2. **分治合并**

```js
var mergeKLists = function(lists) {
    return merge(lists,0,lists.length-1);
};
function merge(lists,l,r){
    if(l===r){
        return lists[l];
    }
    if(l>r){
        return null;
    }
    let mid=(l+r)>>1;
    return mergeTwoLists(merge(lists,l,mid),merge(lists,mid+1,r));
}
let mergeTwoLists = function(a,b){
    if(a===null) return b;
    if(b===null) return a;
    let result = new ListNode(0);
    let temp=result;

            temp.next=a;
            a=a.next;

            b=b.next;
        }
        temp=temp.next;
    }
    if(a===null){
        temp.next=b;
    }else{
        temp.next=a;
    }
    return result.next;
}
```
